% Hack, damit Aufruf von \tableofcontents auch gleich \addcontentsline
% enthält. Damit steht im Inhaltsverzeichnis das Inhaltsverzeichnis
% mit korrekter Seitenzahl und Hyperlink.
% Ohne das ergab sich per:
%   \newpage
%   \addcontentsline{toc}{section}{Inhaltsverzeichnis}  ==> NOT USED
%       --> --- springt auf eine Seite zu früh,
%       --> +   zeigt korrekte Seitenzahl im Inhaltsverzeichnis an. 
%   \tableofcontents
%   \addcontentsline{toc}{section}{Inhaltsverzeichnis}  ==> USED, da Springen
%       --> -   springt eine Zeile zu weit,                       wichtiger.
%       --> -   zeigt S. 8 statt S. 3 (also das Ende des Verzeichnisses)
\makeatletter
\renewcommand\tableofcontents{%
    \section*{\contentsname
        \@mkboth{%
           \MakeUppercase\contentsname}{\MakeUppercase\contentsname}}%
    \addcontentsline{toc}{section}{\contentsname}
    \@starttoc{toc}%
    }
\makeatother
